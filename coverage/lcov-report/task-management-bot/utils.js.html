    static createRateLimiter(maxCalls, timeWindow) {
        const calls = [];
        
        return function() {
            const now = Date.now();
            
            // Remove old calls outside the time window
            while (calls.length > 0 && now - calls[0] > timeWindow) {
                calls.shift();
            }
            
            // Check if we've exceeded the limit
